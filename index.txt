/* ---------- 선생님 구글 시트로 전송 (Apps Script) ---------- */
// ▼▼▼ 선생님: 구글시트 Apps Script 웹앱 주소를 아래에 붙여넣으면 작동합니다 ▼▼▼
const SHEET_URL = 'https://script.google.com/macros/s/AKfycbz2Hp2amHRmW9cNfqG68NKLyaKUwWcAecFNs_XvOEMbALDIM0Aauuzwl_RdNuCRw4z_Rw/exec';  // 구글시트 Apps Script 웹앱
// ▲▲▲ 여기까지 (배포 방법은 선생님 안내문 참고) ▲▲▲
document.getElementById('submitTeacher').onclick=async()=>{
  const msg=document.getElementById('submitMsg');
  if(!chosenTopic){ msg.innerHTML='<span class="no">먼저 위에서 주제를 골라 주세요!</span>'; return; }
  const team=document.getElementById('teamName3').value.trim();
  const s=document.getElementById('roleStart').value.trim();
  const m=document.getElementById('roleMid').value.trim();
  const e=document.getElementById('roleEnd').value.trim();
  if(!team || !s || !m || !e){ msg.innerHTML='<span class="no">모둠 이름과 처음·중간·끝 담당을 모두 적어 주세요!</span>'; return; }
  if(!SHEET_URL){ msg.innerHTML='<span class="no">⚠️ 아직 구글 시트가 연결되지 않았어요.</span>'; return; }

  msg.innerHTML='<span class="status">⏳ 선생님께 보내는 중…</span>';

  // 데이터를 URL 쿼리로 붙여 GET 방식으로 전송 (no-cors POST보다 환경 제약에 강함)
  const params=new URLSearchParams({
    team:team, topic:chosenTopic, start:s, mid:m, end:e,
    time:new Date().toLocaleString('ko-KR')
  });
  const url=SHEET_URL+'?'+params.toString();

  // 방법1: fetch GET (no-cors) — 조용히 전송
  let sent=false;
  try{
    await fetch(url, {method:'GET', mode:'no-cors'});
    sent=true;
  }catch(err){ sent=false; }

  // 방법2: fetch가 막히면 이미지 비콘으로 재시도
  if(!sent){
    try{
      await new Promise((resolve,reject)=>{
        const img=new Image();
        img.onload=()=>resolve();
        img.onerror=()=>resolve(); // Apps Script는 이미지가 아니어도 요청은 도달함
        img.src=url;
        setTimeout(resolve, 1500);
      });
      sent=true;
    }catch(err){ sent=false; }
  }

  if(sent){
    msg.innerHTML='<span class="ok">✅ 제출했어요! (주제: '+chosenTopic+') 선생님 시트를 확인해 주세요.</span>';
    document.getElementById('submitTeacher').disabled=true;
  }else{
    msg.innerHTML='<span class="no">전송이 안 됐어요. 같은 화면이 계속되면, 이 페이지를 실제 인터넷 주소(깃허브 페이지 등)에서 열어야 작동해요.</span>';
  }
};

function buildLog(){
  const team=(document.getElementById('teamName2')?.value.trim())||'우리 모둠';

  let html=`<div class="logteam">📌 ${team}</div>`;

  html+= score.s1 ? `<span class="ok">완료 ✅</span>` : `<span class="status">진행 중</span>`;
  html+=`</div>`;
  // 2단계
  html+=`<div class="logitem"><span class="logstep">②표지어 찾기</span>`;
  html+= (hit===total&&total>0) ? `<span class="ok">표지어 ${hit}/${total} 완료</span>` : `<span class="status">표지어 ${hit}/${total}</span>`;
  html+=`</div>`;
  // 3단계
  html+=`<div class="logitem"><span class="logstep">③문단 쓰기</span>`;
  html+= (typeof chosenTopic!=='undefined' && chosenTopic) ? `<span class="ok">주제: ${esc(chosenTopic)}</span>` : `<span class="status">주제 미선택</span>`;

  html+=`</div>`;
  return html;
}
